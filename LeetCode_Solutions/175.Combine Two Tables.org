SELECT
 Person.firstName firstName, 
 Person.lastName lastName,
 IFNULL(Address.city, null) city,
 IFNULL(Address.state,null) state
 From Person
 LEFT OUTER JOIN Address
 ON Person.personID = Address.personID
 ORDER BY Person.personID;
